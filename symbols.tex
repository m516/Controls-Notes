}





\entry{$H$}{
\begin{itemize}
    \item Hamiltonian matrix (\href{https://en.wikipedia.org/wiki/Hamiltonian_matrix}{W})
    \item Hamiltonian (Hamiltonian mechanics) (\href{https://en.wikipedia.org/wiki/State_space}{W}) \\
        type: $ \mathbb{R}^n \rightarrow \mathbb{R} $
        \begin{itemize}
            \item Assuming discrete time linear system: $ H_k = L(x_k, u_k, k) + p_{k+1}^{T} f(x_k, u_k, k)  $
        \end{itemize} 
            
            
    
\end{itemize} 
}
